function removeDuplicates(str) { var uniques = ""; var duplicates = 0; for (var i = 0; i < str.length; i++) { if (uniques.indexOf(str[i]) == -1) { uniques += str[i]; } else { duplicates++; } } uniques = uniques.replace(/[&\/\\#,_+()$~%.'":*?<>]/g, '').split('').sort().join('') return { uniques, duplicates } } const testString = 'th#elex_ash?'; const expectedOutput = { uniques: 'aehlstx', duplicates: 2 }; console.log(removeDuplicates(testString));
